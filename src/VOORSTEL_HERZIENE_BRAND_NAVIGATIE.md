## Executive Summary

**Probleem:**  
Huidige navigatie maakt geen onderscheid tussen "asset status" (validatie state) en "research progress" (data collection state), waardoor gebruikers niet kunnen zien:
1. Welke assets klaar zijn om te gebruiken
2. Welke research methoden zijn gebruikt per asset
3. Welke assets nog research nodig hebben
4. Waar gaps zitten in research coverage

**Oplossing:**  
**Dual-Layer Status System** met:
- **Layer 1:** Asset Validation Status (4 states)
- **Layer 2:** Research Method Progress (per method tracking)
- **Layer 3:** Research Coverage Indicators (visual badges)

### Data Model:

```typescript
interface BrandAsset {
  id: string;
  name: string;
  status: 'awaiting-research' | 'in-development' | 'ready-to-validate' | 'validated';
  
  // Research tracking
  researchMethods: ResearchMethod[];
  researchCoverage: number; // 0-100%
  
  // Content tracking
  contentSections: ContentSection[];
  artifactsGenerated: number;
  artifactsValidated: number;
  
  // Metadata
  lastUpdated: string;
  validatedBy?: string;
  validatedAt?: string;
}

interface ResearchMethod {
  type: 'workshop' | 'interviews' | 'questionnaire' | 'ai-exploration';
  status: 'not-started' | 'in-progress' | 'completed' | 'cancelled';
  progress?: number; // 0-100 if in-progress
  completedAt?: string;
  metadata: {
    participants?: number;
    sessions?: number;
    responses?: number;
    facilitator?: string;
  };
  artifacts: string[]; // IDs of generated content
}

interface ContentSection {
  id: string;
  title: string;
  content: string;
  status: 'draft' | 'ready-to-validate' | 'validated';
  sources: ResearchMethodReference[]; // Which research generated this
  validatedAt?: string;
  validatedBy?: string;
}
```

### Status Calculation Logic:

```typescript
function calculateAssetStatus(asset: BrandAsset): AssetStatus {
  const anyMethodInProgress = asset.researchMethods.some(m => m.status === 'in-progress');
  const anyMethodCompleted = asset.researchMethods.some(m => m.status === 'completed');
  const allSectionsValidated = asset.contentSections.every(s => s.status === 'validated');
  const anySectionReadyToValidate = asset.contentSections.some(s => s.status === 'ready-to-validate');
  
  if (allSectionsValidated && !anyMethodInProgress) {
    return 'validated';
  }
  
  if (anySectionReadyToValidate) {
    return 'ready-to-validate';
  }
  
  if (anyMethodInProgress || anyMethodCompleted) {
    return 'in-development';
  }
  
  return 'awaiting-research';
}

function calculateResearchCoverage(asset: BrandAsset): number {
  const totalMethods = 4; // Workshop, Interviews, Questionnaire, AI
  const completedMethods = asset.researchMethods.filter(m => m.status === 'completed').length;
  return (completedMethods / totalMethods) * 100;
}
```
